enum ExerciseGroup {
    BACK,
    CHEST,
    BICEP,
    TRICEP,
    SHOULDER,
    LEGS,
    ABS
}

class Exercise {
    id: number;
    name: string;
    group: ExerciseGroup;
}

class Set {
    id: number; foreign key to exercise
    weight: number;
    reps: number;
}

class Workout {
    id: number;
    date: Date;
    sets: Set[]
}

// View API
class WorkoutView {
    id: number;
    date: Date;
    sets: SetView[];
}

class SetView {
    id: number;
    exerciseId: number;
    name: string;
    group: ExerciseGroup;
    weight: number;
    reps: number;
}
